def time_at beat
	index = @list.bisect(right: true) { _1.beat <=> beat }
	raise ArgumentError, 'beat is before the first bpm change' if index < 0
	bpm = @list[index]
	(0...index).sum @offset + (beat - bpm.beat) / bpm.bps do |i|
		bpm = @list[i]
		(@list[i+1].beat - bpm.beat) / bpm.bps
	end
end
