Vector2D = 
Data.define :x, :y do
	def +@
		self
	end

	def -@
		Vector2D.new -x, -y
	end

	def + other
		Vector2D.new x + other.x, y + other.y
	end

	def - other
		Vector2D.new x - other.x, y - other.y
	end

	def * other
		other.is_a?(Vector2D) ? Vector2D.new(x*other.x, y*other.y) : Vector2D.new(x*other, y*other)
	end

	def / other
		other.is_a?(Vector2D) ? Vector2D.new(x/other.x, y/other.y) : Vector2D.new(x/other, y/other)
	end

	def to_a
		[x, y]
	end

	def length
		Math.hypot x, y
	end

	def angle
		Math.atan2 y, x
	end

	def length2
		x*x + y*y
	end

	def rotate angle
		cos = Math.cos angle
		sin = Math.sin angle
		Vector2D.new x*cos - y*sin, x*sin + y*cos
	end

	def self.from_string str
		new *str.split(',').map(&:to_f)
	end

	def self.from_polar radius, angle
		new Math.cos(angle)*radius, Math.sin(angle)*radius
	end
end
